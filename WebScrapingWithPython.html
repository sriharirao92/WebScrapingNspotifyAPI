Here we are scraping the website "techcrunch.com" to retrive all the articles published along with publish date, author and associated tags.

# Import the required packages
from bs4 import BeautifulSoup
import pandas as pd
import requests
import numpy as np

html = requests.get("https://techcrunch.com/").text # send request to retrieve source HTML from the website
soup = BeautifulSoup(html, 'html5lib') # Create a beautiful soup object using the HTML retrieved in previous step
techcrunch = pd.DataFrame() # Create a empty dataframe

post=[]
i = 0
for li in soup.find_all("li", class_="river-block"):
    if (li.find("div", class_="block-content")) is not None:
        post.append(li.find("div", class_="block-content").find("h2").find("a").text.rstrip())
    elif (li.find("div", class_="block-content-brief")) is not None:
        post.append(li.find("div", class_="block-content-brief").find("h2").find("a").text.rstrip())
    else:
        post.append('NA')
    
    if (li.find("div", class_="block-content")) is not None:
        post.append(li.find("div", class_="block-content").find("div", class_="byline").find("time")['datetime'])
    elif (li.find("div", class_="block-content-brief")) is not None:
        post.append(li.find("div", class_="block-content-brief").find("div", class_="byline").find("time")['datetime'])
    else:
        post.append('NA')
        
    if (li.find("div", class_="block-content")) is not None:
        post.append(li.find("div", class_="block-content").find("div", class_="byline").find("a").text.rstrip())
    elif (li.find("div", class_="block-content-brief")) is not None:
        post.append(li.find("div", class_="block-content-brief").find("div", class_="byline").find("a").text.rstrip())
    else:
        post.append('NA')
        
    if (li.find("div", class_="tags")) is not None:
        if (len(li.find("div", class_="tags").findChildren()) > 0):
            post.append(li.find("div", class_="tags").find("a").text.rstrip())
        else:
            post.append('NA')
    else:
        post.append('NA')   
    
    techcrunch = techcrunch.append(post, ignore_index=True)
    i+=1
    post=[]

techcrunch = pd.DataFrame(techcrunch.as_matrix().reshape(i,4))
techcrunch.columns = ['article title','publish date','author','tags']
print('\t\t\t\tArticles posted on TechCrunch.com\n\t\t\t\t*********************************')
techcrunch

Retrieve number of votes and tally from "www.senate.gov" webpage¶

# Import the required packages
from bs4 import BeautifulSoup
import pandas as pd
import requests
import numpy as np
import re


html = requests.get("https://www.senate.gov/legislative/LIS/roll_call_lists/vote_menu_115_1.htm").text # send request to retrieve source HTML from the website
soup = BeautifulSoup(html, 'html5lib') # Create a beautiful soup object using the HTML retrieved in previous step

# Get the HTML 'table' tag containing the required data and iterate through each row in the HTML Table and obtain the number of votes and Tally numbers.  
body = soup.find("tbody")
length = len(body.find_all("tr"))
arr = np.empty([length,3],dtype='a30') # Generate an empty numpy array with character type.

# loop through each row using HTML 'tr' tag and obtain data from first two columns i.e 'Votes' and 'Tally'.
for ind, tr in enumerate(body.find_all("tr")):
    for index,td in enumerate(tr.find_all("td")):
        if index == 0:  # for column 'Votes'
            # scrape the data (Inner HTML) of <a> anchor tag and format the string.
            arr[ind,index] = re.split('\\xa0', td.find("a").text.rstrip())[0]
            a = re.split('\\xa0', td.find("a").text.rstrip())[1].replace('(','').replace(')','')
            # convert the string obtained from tally column and convert it to 'integer' type. Now subtract the two numbers (Voted Yes, Voted No) to obtain the margin 
            arr[ind,index+1] = abs(int(re.split('-',a)[0])-int(re.split('-',a)[1]))
        elif index == 1:
            arr[ind,index+1] = td.text.rstrip()
        
arr = pd.DataFrame(arr)  
arr.columns=['Votes','Tally','Result']
arr = arr[arr.Result == 'Rejected']
arr = arr[['Votes','Tally']].apply(pd.to_numeric)
arr = arr[arr.Tally < 5]
print('Number of U.S. Senate votes in the 115th Congress that were rejected by a margin of less than 5 votes:\n*****************************************************************************************************')
print(str(len(arr))+'\n\n')

arr = arr.reset_index(drop=True)
print('List of votes (#) that were rejected by a margin of less than 5 votes:\n**********************************************************************')
arr

# Import the required packages
from bs4 import BeautifulSoup
import pandas as pd
import requests
import numpy as np
import re


html = requests.get("https://travel.state.gov/content/passports/en/alertswarnings.html").text # send request to retrieve source HTML from the website
soup = BeautifulSoup(html, 'html5lib') # Create a beautiful soup object using the HTML retrieved in previous step
NWarnings = 0
NAlerts = 0

# Get the HTML 'table' tag containing the required data and iterate through each row in the HTML Table and obtain the number of warnings and alerts.  
body = soup.find("tbody")
for ind, tr in enumerate(body.find_all("tr")):
    for index,td in enumerate(tr.find_all("td")):
        if index == 0:
            if (td.text.rstrip()) == 'Warning':
                NWarnings+=1
            elif (td.text.rstrip() == 'Alert'):
                NAlerts+=1

print('Number of alerts and warnings for international travel given by the US government:\n**********************************************************************************')                
print('Number of Warnings = '+str(NWarnings)+'\nNumber of Alerts = '+str(NAlerts))

# Import the required packages
from spotipy.oauth2 import SpotifyClientCredentials
import json
import spotipy
import pandas as pd
import numpy as np
import matplotlib.pyplot as py

# Read the data of songs using Billboard’s Charts from Jan 1, 2000 - Dec. 31, 2005. (provided)
data = pd.read_csv("song-list.csv")
data = data[data.Successful == 1]
finalDataSet = pd.DataFrame()

# Use the Spotify Client Crediantials manager to connect to spotify API using 'client ID' and 'client secret'
client_credentials_manager = SpotifyClientCredentials('4eea203709794ec0a6d3b0b13381b1fb','e0cff7aebe1c41efb114073b8f110b01')
sp = spotipy.Spotify(client_credentials_manager=client_credentials_manager)
sp.trace=False

# For each song in the provided dataset, use 'search' function of the API to query based on 'track name' and 'artist name'
# Traverse the dictionary returned by search operation to find the URI of the track. Using the URI we can get the audio features for each song using the method 'audio_features' of the spotify API.
for ind,eachRow in enumerate(data.values):
    results = sp.search(q='track:"'+eachRow[0]+'"+artist:"'+eachRow[1]+'"')
    if (results['tracks']['items'] != []):
        finalDataSet = finalDataSet.append(pd.DataFrame(sp.audio_features(str(results['tracks']['items'][0]['uri']))[0], index=[0]))

# Create a scatterplot of Energy Vs Valence for all the songs in the dataset provided.
print('Done fetching audio features for all the songs\n\n')
py.figure(figsize=(15,10))
py.title("Energy Vs Valence for songs", fontsize=16) # Set the plot title
py.ylabel("Valence",fontsize=14) # Set the Y-Label
py.xlabel("Energy",fontsize=14) # Set the X-Label
py.scatter(finalDataSet.energy,finalDataSet.valence)
py.show()
